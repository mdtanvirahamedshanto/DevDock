// Fetch latest release from GitHub API and Auto-Detect OS document.addEventListener('DOMContentLoaded', async () => { const GITHUB_OWNER = 'mdtanvirahamedshanto'; const GITHUB_REPO = 'DevDock'; // 1. Detect OS let userOS = "Unknown OS"; let osType = "none"; if (navigator.userAgent.indexOf("Win") != -1) { userOS = "Windows"; osType = "win"; } else if (navigator.userAgent.indexOf("Mac") != -1) { userOS = "macOS"; osType = "mac"; } else if (navigator.userAgent.indexOf("Linux") != -1) { userOS = "Linux"; osType = "linux"; } // 2. Update Hero Button Text if (osType !== "none") { document.getElementById('hero-os-name').innerText = userOS; } // 3. Fetch from GitHub try { const response = await fetch(`https://api.github.com/repos/${GITHUB_OWNER}/${GITHUB_REPO}/releases/latest`); const data = await response.json(); // Update version text everywhere const versionElements = document.querySelectorAll('.release-version'); versionElements.forEach(el => el.innerText = data.tag_name || "Latest"); if (data.assets) { const macAsset = data.assets.find(a => a.name.endsWith('.dmg')); const winAsset = data.assets.find(a => a.name.endsWith('.exe')); const linuxAsset = data.assets.find(a => a.name.endsWith('.AppImage')); const debAsset = data.assets.find(a => a.name.endsWith('.deb')); // Update cards if (macAsset) document.getElementById('card-mac-dmg').href = macAsset.browser_download_url; if (winAsset) document.getElementById('card-win-exe').href = winAsset.browser_download_url; if (linuxAsset) document.getElementById('card-linux-appimage').href = linuxAsset.browser_download_url; if (debAsset) document.getElementById('card-linux-deb').href = debAsset.browser_download_url; // Update Hero Button Link based on OS const heroBtn = document.getElementById('hero-download-btn'); if (osType === 'mac' && macAsset) heroBtn.href = macAsset.browser_download_url; else if (osType === 'win' && winAsset) heroBtn.href = winAsset.browser_download_url; else if (osType === 'linux' && linuxAsset) heroBtn.href = linuxAsset.browser_download_url; } } catch (error) { console.error('Failed to fetch latest releases from GitHub:', error); } });
